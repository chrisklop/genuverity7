        // 8. INTERACTION HANDLERS

        function bindTextHighlighting() {

            const article = document.getElementById('report-content');

            const menu = document.getElementById('context-menu');

            article.addEventListener('mouseup', () => {

                const selection = window.getSelection();

                if (selection.toString().length > 3) {

                    const range = selection.getRangeAt(0);

                    const rect = range.getBoundingClientRect();

                    menu.style.top = `${rect.bottom + 10}px`;

                    menu.style.left = `${rect.left + (rect.width/2) - 60}px`;

                    menu.classList.add('visible');

                    window.currentSelection = selection.toString();

                } else {

                    menu.classList.remove('visible');

                }

            });

            document.addEventListener('mousedown', (e) => {

                if(!menu.contains(e.target)) menu.classList.remove('visible');

            });

        }
